# created submodule for vulkan docs
git submodule add --depth=1 git@github.com:KhronosGroup/Vulkan-Docs.git submodules/vulkandocs
# make sure to get all the tags of the submodule as well
git fetch --tags --recurse-submodules=yes


# generate vulkan_core.h C header
python3 submodules/Vulkan-Docs/scripts/genvk.py vulkan_core.h -registry submodules/vulkandocs//xml/vk.xml

# bash get vk version string



## GIT ##
# make sure to use recurse when switching branches, because submodule dirs in branches will stay there when switching to master 
git checkout --recurse-submodules master

# git setup a way to get the tags of a submodule from the parent project
#https://stackoverflow.com/questions/62912975/git-submodules-how-to-know-the-commit-tag-of-the-submodule

# To turn a hash ID into a tag name
#  Note that you must run this Git command in the submodule, after using the superproject to find the hash ID
git tag --points-at <hash>
